Daemon settings (ccpyd.conf)

Example config file

<ccpyd>
  <ccpyConfig>/etc/ccpy.conf</ccpyConfig>
  <schedule time="06:30"/>
  <logging enabled="on">
    <file>/var/log/ccpyd.log</file>
    <level>DEBUG</level>
  </logging>
</ccpyd>
